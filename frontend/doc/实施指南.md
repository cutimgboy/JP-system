    defaultNS: 'common', // 默认命名空间

      order: ['localStorage', 'navigator', 'htmlTag'],
      caches: ['localStorage'],
      lookupLocalStorage: 'i18nextLng',
    },
    
    // 其他配置
    interpolation: {
      escapeValue: false, // React 已经转义
    },
    react: {
      useSuspense: false, // SSR 不支持 Suspense

### 2.4 创建 HTTP 客户端

**文件**: `app/lib/api/client.ts`

```typescript
import axios, { AxiosInstance, AxiosRequestConfig, AxiosResponse } from 'axios';
import type { ApiResponse, ApiError, RequestConfig } from './types';

// 创建 axios 实例
const apiClient: AxiosInstance = axios.create({
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// 请求拦截器
apiClient.interceptors.request.use(
  (config) => {
    // 添加认证 Token
    const token = localStorage.getItem('token');
    if (token && !(config as any).skipAuth) {
      config.headers.Authorization = `Bearer ${token}`;
    }

    // 添加语言标识
    const language = localStorage.getItem('i18nextLng') || 'zh-CN';
    config.headers['Accept-Language'] = language;

    // 添加请求 ID（用于日志追踪）
    config.headers['X-Request-ID'] = `${Date.now()}-${Math.random()}`;

    return config;
  },
  (error) => {
    return Promise.reject(error);
  }
);

// 响应拦截器
apiClient.interceptors.response.use(
  (response: AxiosResponse<ApiResponse>) => {
    const { data } = response;
    
    // 如果后端返回的是包装格式 { code, message, data }
    if (data.code !== undefined) {
      if (data.code === 200 || data.code === 0) {
        return data.data; // 返回实际数据
      } else {
        // 业务错误
        const error: ApiError = {
          code: data.code,
          message: data.message || '请求失败',
        };
        return Promise.reject(error);
      }
    }
    
    // 如果后端直接返回数据
    return data;
  },
  (error) => {
    // 统一错误处理
    if (error.response) {
      const { status, data } = error.response;
      
      const apiError: ApiError = {
        code: status,
        message: data?.message || getErrorMessage(status),
        details: data,
      };

      // 处理特定状态码
      if (status === 401) {
        // 未授权，清除 Token 并跳转登录
        localStorage.removeItem('token');
        window.location.href = '/auth/login';
      } else if (status === 403) {
        // 无权限
        console.error('无权限访问');
      } else if (status === 404) {
        // 资源不存在
        console.error('资源不存在');
      } else if (status >= 500) {
        // 服务器错误
        console.error('服务器错误');
      }

      return Promise.reject(apiError);
    } else if (error.request) {
      // 网络错误
      const apiError: ApiError = {
        code: 0,
        message: '网络连接失败，请检查网络设置',
      };
      return Promise.reject(apiError);
    } else {
      // 其他错误
      return Promise.reject(error);
    }
  }
);

// 获取错误消息
function getErrorMessage(status: number): string {
  const errorMap: Record<number, string> = {
    400: '请求参数错误',
    401: '未授权，请先登录',
    403: '无权限访问',
    404: '资源不存在',
    500: '服务器错误，请稍后重试',
    502: '网关错误',
    503: '服务不可用',
  };
  return errorMap[status] || '请求失败';
}

// 导出封装的请求方法
export const api = {
  get: <T = any>(url: string, config?: AxiosRequestConfig & RequestConfig) => {
    return apiClient.get<T>(url, config);
  },
  post: <T = any>(url: string, data?: any, config?: AxiosRequestConfig & RequestConfig) => {
    return apiClient.post<T>(url, data, config);
  },
  put: <T = any>(url: string, data?: any, config?: AxiosRequestConfig & RequestConfig) => {
    return apiClient.put<T>(url, data, config);
  },
  delete: <T = any>(url: string, config?: AxiosRequestConfig & RequestConfig) => {
    return apiClient.delete<T>(url, config);
  },
  patch: <T = any>(url: string, data?: any, config?: AxiosRequestConfig & RequestConfig) => {
    return apiClient.patch<T>(url, data, config);
  },
};

export default apiClient;
```

### 2.6 创建 useApi Hook

**文件**: `app/hooks/useApi.ts`

```typescript
import { useState, useEffect, useCallback } from 'react';
import type { ApiError } from '~/lib/api/types';

interface UseApiOptions<T> {
  immediate?: boolean; // 是否立即执行
  onSuccess?: (data: T) => void;
  onError?: (error: ApiError) => void;
}

interface UseApiResult<T> {
  data: T | null;
  loading: boolean;
  error: ApiError | null;
  execute: (...args: any[]) => Promise<T | undefined>;
  refetch: () => Promise<T | undefined>;
}

export function useApi<T = any>(
  apiFunction: (...args: any[]) => Promise<T>,
  deps: any[] = [],
  options: UseApiOptions<T> = {}
): UseApiResult<T> {
  const { immediate = true, onSuccess, onError } = options;
  
  const [data, setData] = useState<T | null>(null);
  const [loading, setLoading] = useState(immediate);
  const [error, setError] = useState<ApiError | null>(null);

  const execute = useCallback(
    async (...args: any[]): Promise<T | undefined> => {
      setLoading(true);
      setError(null);
      
      try {
        const result = await apiFunction(...args);
        setData(result);
        onSuccess?.(result);
        return result;
      } catch (err) {
        const apiError = err as ApiError;
        setError(apiError);
        onError?.(apiError);
        return undefined;
      } finally {
        setLoading(false);
      }
    },
    [apiFunction, onSuccess, onError]
  );

  const refetch = useCallback(() => {
    return execute();
  }, [execute]);

  useEffect(() => {
    if (immediate) {
      execute();
    }
  }, [immediate, ...deps]);

  return {
    data,
    loading,
    error,
    execute,
    refetch,
  };
}
```

import { redirect } from 'react-router';

export async function requireAuth(request: Request): Promise<Response | null> {
  // 从 Cookie 或 Header 获取 Token
  const token = request.headers.get('Authorization')?.replace('Bearer ', '') ||
                new URL(request.url).searchParams.get('token') ||
                null;

  if (!token) {
    // 未认证，重定向到登录页
    const url = new URL(request.url);
    throw redirect(`/auth/login?redirect=${encodeURIComponent(url.pathname)}`);
  }

  // 可选：验证 Token 有效性
  // const isValid = await validateToken(token);
  // if (!isValid) {
  //   throw redirect('/auth/login');
  // }

  return null;
}

import { Form, useActionData, useNavigation } from 'react-router';
import { userApi } from '~/lib/api/endpoints/user';
import type { Route } from './+types/login';

export const meta: Route.MetaFunction = () => {
  return [
    { title: "登录" },
    { name: "description", content: "用户登录页面" },
  ];
};

export const action: Route.ActionFunction = async ({ request }) => {
  const formData = await request.formData();
  const email = formData.get('email') as string;
  const password = formData.get('password') as string;

  try {
    const response = await userApi.login({ email, password });
    // 保存 Token
    localStorage.setItem('token', response.token);
    
    // 重定向到目标页面或首页
    const redirect = new URL(request.url).searchParams.get('redirect') || '/';
    return redirect(redirect);
  } catch (error) {
    return { error: '登录失败，请检查用户名和密码' };
  }
};

export default function Login() {
  const actionData = useActionData<typeof action>();
  const navigation = useNavigation();
  const isSubmitting = navigation.state === 'submitting';

  return (
    <div className="bg-white p-8 rounded-lg shadow-md">
      <h1 className="text-2xl font-bold mb-6">登录</h1>
      <Form method="post">
        <div className="mb-4">
          <label className="block text-sm font-medium mb-2">邮箱</label>
          <input
            type="email"
            name="email"
            required
            className="w-full px-3 py-2 border rounded"
          />
        </div>
        <div className="mb-4">
          <label className="block text-sm font-medium mb-2">密码</label>
          <input
            type="password"
            name="password"
            required
            className="w-full px-3 py-2 border rounded"
          />
        </div>
        {actionData?.error && (
          <div className="mb-4 text-red-600 text-sm">{actionData.error}</div>
        )}
        <button
          type="submit"
          disabled={isSubmitting}
          className="w-full bg-blue-600 text-white py-2 rounded hover:bg-blue-700 disabled:opacity-50"
        >
          {isSubmitting ? '登录中...' : '登录'}
        </button>
      </Form>
    </div>
  );
}

## 六、常见问题

### Q1: SSR 环境下 i18n 如何初始化？
A: 在服务端，从请求头 `Accept-Language` 获取语言，初始化 i18n 实例。

### Q2: 如何避免 API 请求在 SSR 时暴露敏感信息？
A: 使用环境变量区分服务端和客户端 API 地址，服务端使用内网地址。

### Q3: 路由守卫在 SSR 时如何工作？
A: 在路由的 `loader` 函数中检查认证状态，未认证时抛出 `redirect`。
